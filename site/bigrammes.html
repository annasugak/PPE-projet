1 - Script pour l'Anglais

#!/bin/bash
mkdir -p ./bigrammes/bigrammes_anglais

for fichier in ./contextes/contextes_anglais/*.txt
do
    [ -e "$fichier" ] || continue
    nom_base=$(basename "$fichier" .txt)
    fichier_sortie="./bigrammes/bigrammes_anglais/bigrammes-${nom_base}.html"

    echo "" > "$fichier_sortie"
    echo "
Bigrammes pour : $nom_base
" >> "$fichier_sortie"
    echo "" >> "$fichier_sortie"
    echo "" >> "$fichier_sortie"

    tr -cs "[:alpha:]" "\n" < "$fichier" > contexte_au_propre.txt

    paste -d " " contexte_au_propre.txt <(tail -n +2 contexte_au_propre.txt) | \
    sort | uniq -c | sort -nr | head -n 20 | \
    awk '{print ""}' >> "$fichier_sortie"

    echo "
FréquenceBigramme
" $1 "" $2 " " $3 "
" >> "$fichier_sortie"
done
rm contexte_au_propre.txt

3 - Script pour le Russe

#!/bin/bash
mkdir -p ./bigrammes/bigrammes_russe
compteur=1

for fichier in ./contextes/contextes_russe/*.txt
do
    [ -e "$fichier" ] || continue
    nom_base=$(basename "$fichier" .txt)
    fichier_sortie="./bigrammes/bigrammes_russe/bigrammes_contextes_russe_${compteur}.html"

    echo "" > "$fichier_sortie"
    echo "
Bigrammes pour le mot 'свет' : $nom_base
" >> "$fichier_sortie"
    echo "" >> "$fichier_sortie"
    echo "" >> "$fichier_sortie"

    grep -oE "[а-яА-ЯёЁa-zA-Z]+" "$fichier" > contexte_au_propre.txt

    paste -d " " contexte_au_propre.txt <(tail -n +2 contexte_au_propre.txt) | \
    grep -i "свет" | \
    sort | uniq -c | sort -nr | head -n 20 | \
    awk '{print ""}' >> "$fichier_sortie"

    echo "
FréquenceBigramme
" $1 "" $2 " " $3 "
" >> "$fichier_sortie"
    ((compteur++))
done
rm -f contexte_au_propre.txt
