            if (!armed) return;

            document.removeEventListener('pointerdown', onUserIntent);
            document.removeEventListener('click', onUserIntent);

            loadAndPlay();
          }
          function onUserIntent() {
            loadAndPlay();

            const keys = ['ArrowDown', 'PageDown', 'Space', 'End', 'Home'];

          });

            passive: true,
